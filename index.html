const express = require("express"); const cors = require("cors"); const TelegramBot = require("node-telegram-bot-api"); const app = express(); app.use(cors()); app.use(express.json()); // ===================== CONFIG ===================== const BOT_TOKEN = "PUT_YOUR_BOT_TOKEN_HERE"; // 👈 هنا التوكن const ADMIN_ID = "7771042305"; // ================================================== const bot = new TelegramBot(BOT_TOKEN, { polling: true }); // ===================== DATA ===================== let countries = [ { name: "مصر", key: "egypt" }, { name: "السعودية", key: "ksa" }, { name: "أمريكا", key: "usa" } ]; let numbers = { egypt: [ { number: "+20 100 111 2222", price: "50 EGP" } ], ksa: [ { number: "+966 55 123 4567", price: "5 SAR" } ], usa: [ { number: "+1 202 555 0101", price: "2 USD" } ] }; let orders = []; // ===================== TELEGRAM ===================== function sendOrder(order){ bot.sendMessage(ADMIN_ID, ` 🔥 NEW ORDER VIP 🌍 Country: ${order.country} 📱 Number: ${order.number} 💰 Payment: ${order.payment} 👤 User: ${order.user} `); } // ===================== API ===================== app.get("/api/countries",(req,res)=>res.json(countries)); app.get("/api/numbers/:key",(req,res)=>{ res.json(numbers[req.params.key] || []); }); app.post("/api/order",(req,res)=>{ const order = req.body; orders.push(order); sendOrder(order); res.json({ok:true}); }); // ===================== ADMIN (hidden endpoint) ===================== app.get("/admin-data",(req,res)=>{ if(req.query.id !== ADMIN_ID) return res.send("no access"); res.json({countries,numbers,orders}); }); // ===================== FRONTEND ===================== app.get("/",(req,res)=>{ res.send(` new number VIP

`); }); // ===================== START ===================== app.listen(3000,()=>console.log("VIP SYSTEM RUNNING 🔥"));
